Dart(9)——随机数

@LM | 2021-07-23

1.随机数

import 'dart:math';
main(){
//实例化Random类并赋值给变量rng；
  var rng = new Random();
//打印变量 rng，随机数范围(0-99);
  print(rng.nextInt(100));
}

2.Random

//返回的0~(max-1)的随机数
int nextInt(int max);
//生成 1.0~0.0 的随机数
double nextDouble();
//生成随机布尔值: true/false
bool nextBool();
